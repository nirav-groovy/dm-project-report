+   * Enriches outbound SMS records with replyStatus (Replied, Pending, NoReply).

+  private async enrichWithReplyStatus(data: any[]): Promise<any[]> {
+    const outboundSms = data.filter((d) => d.direction === DirectionType.outbound);
+    if (outboundSms.length === 0) return data;
+
+    const patientPhones = [...new Set(outboundSms.map((d) => d.to).filter((p): p is string => Boolean(p)))];
+    if (patientPhones.length === 0) return data;
+
+    // For each phone, find the overall latest outbound sent_at and latest inbound received_at
+    const latestInteractions = await CommunicationHistory.aggregate([
+      {
+        $match: {
+          $or: [
+            { to: { $in: patientPhones }, direction: DirectionType.outbound, type: CommunicationType.sms },
+            { from: { $in: patientPhones }, direction: DirectionType.inbound, type: CommunicationType.sms },
+          ],
+        },
+      },
+      {
+        $group: {
+          _id: {
+            phone: { $cond: [{ $eq: ["$direction", DirectionType.outbound] }, "$to", "$from"] },
+          },
+          latestOutboundSentAt: {
+            $max: { $cond: [{ $eq: ["$direction", DirectionType.outbound] }, "$sent_at", null] },
+          },
+          latestInboundReceivedAt: {
+            $max: { $cond: [{ $eq: ["$direction", DirectionType.inbound] }, "$received_at", null] },
+          },
+        },
+      },
+    ]);
+
+    const latestInteractionMap = new Map(latestInteractions.map((li: any) => [li._id.phone, li]));
+
+    const now = moment();
+    const pendingThreshold = params.sms_reply_status.pending_threshold_hours;
+    const replyWindow = params.sms_reply_status.reply_window_days;
+
+    return data.map((item) => {
+      if (item.direction !== DirectionType.outbound || item.type !== CommunicationType.sms) {
+        return item;
+      }
+
+      const phone = item.to;
+      const interactions = latestInteractionMap.get(phone);
+
+      let replyStatus = "NoReply"; // Default
+
+      if (interactions) {
+        const itemSentAt = moment(item.sent_at);
+        const latestOutboundTime = moment(interactions.latestOutboundSentAt);
+        const isMostRecentOutbound = itemSentAt.isSame(latestOutboundTime);
+        const latestInboundTime = interactions.latestInboundReceivedAt ? moment(interactions.latestInboundReceivedAt) : null;
+
