+import { FollowUp } from "../../models/mongodb/follow_up";
+import { FollowUpStatus } from "../../types/follow_up";
+import { ServiceBusUtils } from "../../utils/serviceBus";
+import appLogger from "../../utils/logger";
+
+const env = process.env.NODE_ENV || "development";
+const NOTIFICATION_QUEUE = `${env}_services_communication`;
+
+export async function followupNotificationHandler(message: any): Promise<void> {
+  appLogger.info("Handling Followup Notification Message", message.body);
+  const serviceBusUtils = ServiceBusUtils.getInstance();
+
+  try {
+    if (!message.body || typeof message.body !== "object") {
+      appLogger.error("Invalid message format: No body found.");
+      return;
+    }
+
+    if (typeof message.body.type !== "string") {
+      appLogger.error("Invalid message format: Type is missing or not a string.");
+      return;
+    }
+
+    if (typeof message.body.data !== "object") {
+      appLogger.error("Invalid message data format: Data is missing or not an object.");
+      return;
+    }
+
+    const eventData = message.body.data;
+
+    if (typeof eventData.event !== "string") {
+      appLogger.error("Invalid event data: event name is missing or not a string.");
+      return;
+    }
+
+    const { event, followupId, recruiterEmail, patientName, dueDate } = eventData;
+
+    const followup = await FollowUp.findById(followupId);
+
+    if (!followup) {
+      appLogger.warn(`Followup ${followupId} not found`);
+      return;
+    }
+
+    if (followup.status !== FollowUpStatus.UPCOMING) {
+      appLogger.info(`Followup ${followupId} status is ${followup.status}, skipping notification`);
+      return;
+    }
+
+    switch (event) {
+      case "followup-reminder-15min":
+        await sendSocketNotificationEvent(recruiterEmail, patientName, dueDate, "15min", serviceBusUtils);
+        break;
+      case "followup-reminder-overdue":
+        await sendSocketNotificationEvent(recruiterEmail, patientName, dueDate, "overdue", serviceBusUtils);
+        break;
+      default:
+        appLogger.warn(`No handler found for followup event: ${event}`);
+    }
+  } catch (error) {
+    appLogger.error("Error processing followup notification message:", error);
+  }
+}
+
+async function sendSocketNotificationEvent(
+  recruiterEmail: string,
+  patientName: string,
+  dueDate: string,
+  reminderType: "15min" | "overdue",
+  serviceBusUtils: ServiceBusUtils
+): Promise<void> {
+  try {
+    console.log("[FOLLOWUP] Patient Service bus received", reminderType);
+
+    const dueDateTime = new Date(dueDate).toLocaleString();
+    let socketEvent = "followup-reminder";
+    let title = "Follow-up Reminder";
+    let content = `📞 Reminder: You have a follow-up call with ${patientName} scheduled for ${dueDateTime} in 15 minutes`;
+
+    if (reminderType === "overdue") {
+      socketEvent = "followup-overdue";
+      title = "Follow-up Overdue";
+      content = `⚠️ Overdue: Follow-up call with ${patientName} (due ${dueDateTime}) is overdue — status: upcoming`;
+    }
+
+    const socketNotificationMessage = {
+      body: {
+        type: "notificationHandler",
+        data: {
+          event: "send-socket-notification",
+          data: {
+            userEmail: recruiterEmail,
+            socketEvent,
+            type: "followup",
+            socketValue: {
+              title,
+              content,
+              type: reminderType === "15min" ? "reminder" : "overdue",
+              dueDate,
+              patientName,
+            },
+          },
+        },
+      },
+    };
+    console.log("[FOLLOWUP] Notification Service bus called", JSON.stringify(socketNotificationMessage));
+
+    await serviceBusUtils.sendMessage(NOTIFICATION_QUEUE, socketNotificationMessage);
+    appLogger.info(`Socket notification event sent for ${reminderType} reminder to ${recruiterEmail}`);
+  } catch (error) {
+    appLogger.error("Error sending socket notification event:", error);
+  }
+}
