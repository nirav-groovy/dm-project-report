+  private async fetchAndCacheAll(patientId: number) {
+    let comtrakAvailable = true;
+    let commoduleAvailable = true;
+    let comtrakItems: ComtrakInteractionItem[] = [];

+    const [comtrakResult, commModuleResult] = await Promise.allSettled([

+    if (comtrakResult.status === "fulfilled") {

+    } catch (cacheError: any) {

+    }
+
+    return { merged, cachedAt, comtrakAvailable, commoduleAvailable };

+  async getUnifiedHistory(

+    userEmail: string,
+    pageNo: number,
+    pageSize: number,
+    refreshCache: boolean = false
+  ) {
+    try {
+      const cacheKey = `unified:${patientId}`;
+      let comtrakAvailable = true;
+      let commoduleAvailable = true;
+      let allItems: UnifiedHistoryItem[];
+      let cachedAt: string | null = null;
+      let fromCache = false;
+
+      const cached = refreshCache ? null : await this.commonUtils.getCache(cacheKey);
+
+      if (cached?.data && cached?.cachedAt) {
+        allItems = cached.data as UnifiedHistoryItem[];
+        cachedAt = cached.cachedAt;
+        fromCache = true;
+      } else {
+        const result = await this.fetchAndCacheAll(patientId);
+        allItems = result.merged;
+        cachedAt = result.cachedAt;
+        comtrakAvailable = result.comtrakAvailable;
+        commoduleAvailable = result.commoduleAvailable;
+      }
+
+      // Apply "my" filter in-memory (no separate cache needed)
+      let filtered = allItems;
+      if (filter === "my" && userEmail) {
+        const lowerEmail = userEmail.toLowerCase();
+        filtered = allItems.filter((item) =>
+          item.performedByEmail?.toLowerCase() === lowerEmail
+        );
+      }
+
+      const totalCount = filtered.length;
+      const start = (pageNo - 1) * pageSize;
+      const paginatedData = filtered.slice(start, start + pageSize);
+      const pagination = await this.commonUtils.pagination(totalCount, pageNo, pageSize);
+
+      const cacheExpiresAt = cachedAt
+        ? new Date(new Date(cachedAt).getTime() + CACHE_TTL_SECONDS * 1000).toISOString()
+        : null;
+
+      return {
+        status: true,
+        data: paginatedData,
+        pagination,
+        comtrakAvailable,
+        commoduleAvailable,
+        cache: {
+          fromCache,
+          cachedAt,
+          expiresAt: cacheExpiresAt,
+          ttl: CACHE_TTL_SECONDS,
+        },
+      };
+    } catch (error: any) {
+      appLogger.info("UnifiedHistoryService/getUnifiedHistory ::: ", error?.message);
+      throw error;
+    }
+  }
