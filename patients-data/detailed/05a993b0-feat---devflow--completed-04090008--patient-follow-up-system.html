+import { Response } from "express";
+import { UserRequest } from "../types/request";
+import { FollowUp } from "../models/mongodb/follow_up";
+import { FollowUpStatus } from "../types/follow_up";
+import appLogger from "../utils/logger";

+/**
+ * Controller class responsible for managing patient follow-ups.
+ */
+export class FollowUpController {
+  private static instance: FollowUpController | null = null;
+
+  public static getInstance() {
+    FollowUpController.instance ??= new FollowUpController();
+    return FollowUpController.instance;
+  }

+  /**
+   * Create a new follow-up record.
+   * POST /follow-up
+   * Body: { patientId, followUpDate, notes? }
+   */
+  createFollowUp = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { patientId, followUpDate, notes } = req.body;
+      if (!patientId || !followUpDate) {
+        return res.status(400).json({
+          status: false,
+          message: "patientId and followUpDate are required",
+        });
+      }
+
+      const parsedDate = new Date(followUpDate);
+      if (isNaN(parsedDate.getTime())) {
+        return res.status(400).json({ status: false, message: "Invalid followUpDate format" });
+      }
+
+      const followUp = await FollowUp.create({
+        patientId,
+        recruiterId: userId,
+        followUpDate: parsedDate,
+        status: FollowUpStatus.UPCOMING,
+        notes: notes || "",
+        createdBy: userId,
+      });
+
+      return res.status(201).json({ status: true, data: followUp });
+    } catch (error) {
+      appLogger.error("FollowUpController/createFollowUp error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+
+  /**
+   * Get follow-ups for the logged-in user with optional filtering.
+   * GET /follow-up/list?filter=upcoming|overdue|all&patientId=123&page=1&limit=20
+   */
+  getFollowUps = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const {
+        filter = "all",
+        patientId,
+        page = "1",
+        limit = "20",
+      } = req.query as Record<string, string>;
+
+      const pageNum = Math.max(1, parseInt(page));
+      const limitNum = Math.min(100, Math.max(1, parseInt(limit)));
+      const skip = (pageNum - 1) * limitNum;
+      const now = new Date();
+
+      const query: any = { recruiterId: userId };
+
+      if (patientId) {
+        query.patientId = parseInt(patientId);
+      }
+
+      if (filter === "upcoming") {
+        query.followUpDate = { $gte: now };
+        query.status = FollowUpStatus.UPCOMING;
+      } else if (filter === "overdue") {
+        query.followUpDate = { $lt: now };
+        query.status = FollowUpStatus.UPCOMING;
+      }
+
+      const [data, total] = await Promise.all([
+        FollowUp.find(query).sort({ followUpDate: 1 }).skip(skip).limit(limitNum),
+        FollowUp.countDocuments(query),
+      ]);
+
+      return res.status(200).json({
+        status: true,
+        data,
+        pagination: {
+          page: pageNum,
+          limit: limitNum,
+          total,
+          totalPages: Math.ceil(total / limitNum),
+        },
+      });
+    } catch (error) {
+      appLogger.error("FollowUpController/getFollowUps error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+
+  /**
+   * Update a follow-up record (status, notes, reschedule).
+   * PUT /follow-up/:id
+   * Body: { status?, notes?, followUpDate? (for reschedule) }
+   */
+  updateFollowUp = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { id } = req.params;
+      const { status, notes, followUpDate } = req.body;
+
+      const existing = await FollowUp.findOne({ _id: id, recruiterId: userId });
+      if (!existing) {
+        return res.status(404).json({ status: false, message: "Follow-up not found" });
+      }
+
+      const updateData: any = {};
+
+      if (status) {
+        const validStatuses = Object.values(FollowUpStatus);
+        if (!validStatuses.includes(status)) {
+          return res.status(400).json({
+            status: false,
+            message: `status must be one of: ${validStatuses.join(", ")}`,
+          });
+        }
+        updateData.status = status;
+        if (status === FollowUpStatus.COMPLETED) {
+          updateData.completedAt = new Date();
+        }
+      }
+
+      if (notes !== undefined) {
+        updateData.notes = notes;
+      }
+
+      // Handle reschedule: create new follow-up linked to original
+      if (status === FollowUpStatus.RESCHEDULED && followUpDate) {
+        const parsedDate = new Date(followUpDate);
+        if (isNaN(parsedDate.getTime())) {
+          return res.status(400).json({ status: false, message: "Invalid followUpDate format" });
+        }
+
+        // Mark original as rescheduled
+        await FollowUp.findByIdAndUpdate(id, { $set: { status: FollowUpStatus.RESCHEDULED } });
+
+        // Create new follow-up linked to the original
+        const newFollowUp = await FollowUp.create({
+          patientId: existing.patientId,
+          recruiterId: userId,
+          followUpDate: parsedDate,
+          status: FollowUpStatus.UPCOMING,
+          notes: notes || existing.notes || "",
+          createdBy: userId,
+          rescheduledFrom: id,
+        });
+
+        return res.status(200).json({
+          status: true,
+          data: newFollowUp,
+          message: "Follow-up rescheduled successfully",
+        });
+      }
+
+      const updated = await FollowUp.findByIdAndUpdate(id, { $set: updateData }, { new: true });
+      return res.status(200).json({ status: true, data: updated });
+    } catch (error) {
+      appLogger.error("FollowUpController/updateFollowUp error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+
+  /**
+   * Delete a follow-up record.
+   * DELETE /follow-up/:id
+   */
+  deleteFollowUp = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { id } = req.params;
+      const result = await FollowUp.findOneAndDelete({ _id: id, recruiterId: userId });
+
+      if (!result) {
+        return res.status(404).json({ status: false, message: "Follow-up not found" });
+      }
+
+      return res.status(200).json({ status: true, message: "Follow-up deleted successfully" });
+    } catch (error) {
+      appLogger.error("FollowUpController/deleteFollowUp error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+}

+/**

+  {

+      required: true,
+      type: Number,
+      index: true,
+    },
+    recruiterId: {
+      required: true,
+      type: Number,
+      index: true,
+    },
+    followUpDate: {
+      required: true,
+      type: Date,
+      index: true,
+    },
+    status: {
+      required: true,
+      type: String,
+      enum: Object.values(FollowUpStatus),
+      default: FollowUpStatus.UPCOMING,
+    },
+    notes: {
+      required: false,
+      type: String,
+    },
+    createdBy: {
+      required: true,
+      type: Number,
+    },
+    rescheduledFrom: {
+      required: false,
+      type: Schema.Types.ObjectId,
+      ref: "FollowUp",
+    },
+    completedAt: {
+      required: false,
+      type: Date,
+    },
+  },
+  {
+    versionKey: false,
+    collection: "follow_ups",
+    timestamps: true,
+  }

+
+export const FollowUp: Model<IFollowUp> = model<IFollowUp>("FollowUp", FollowUpSchema);
