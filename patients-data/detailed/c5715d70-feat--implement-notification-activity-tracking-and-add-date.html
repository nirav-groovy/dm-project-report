+  /**
+   * POST /message-conversations
+   * Returns SMS conversations grouped by patient (WhatsApp-style chat list).
+   * Each entry has patient info, last message, and upcoming follow-up if any.
+   */
+  getConversations = async (req: UserRequest, res: Response) => {
+    try {
+      const userEmail = req.user?.Email;
+      if (!userEmail) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { search, page = 1, limit = 20 } = req.body;
+
+      const pageNum = Math.max(1, Number(page));
+      const limitNum = Math.min(100, Math.max(1, Number(limit)));
+      const skip = (pageNum - 1) * limitNum;
+
+      // Build match stage
+      const matchStage: any = {
+        user: userEmail,
+        type: CommunicationType.sms,
+        "patient_obj.Id": { $exists: true, $ne: null },
+      };
+
+      // Build search conditions
+      if (search?.trim()) {
+        const searchStr = search.trim();
+        const searchRegex = new RegExp(searchStr, "i");
+        const [firstname, lastname] = searchStr.split(" ");
+        const firstnameRegex = firstname ? new RegExp(`^${firstname}`, "i") : searchRegex;
+        const lastnameRegex = lastname ? new RegExp(`^${lastname}`, "i") : searchRegex;
+
+        const parsedId = Number.parseInt(searchStr);
+        const orConditions: any[] = [
+          { "patient_obj.FirstName": firstnameRegex },
+          { "patient_obj.LastName": lastnameRegex },
+          { to: searchRegex },
+          { from: searchRegex },
+        ];
+
+        if (!Number.isNaN(parsedId)) {
+          orConditions.push({ "patient_obj.Id": parsedId });
+        }
+
+        matchStage.$or = orConditions;
+      }
+
+      // Aggregation pipeline: group SMS by patient, get latest message per patient
+      const pipeline: any[] = [
+        { $match: matchStage },
+        { $sort: { sent_at: -1 } },
+        {
+          $group: {
+            _id: "$patient_obj.Id",
+            patientName: {
+              $first: {
+                $cond: {
+                  if: { $ifNull: ["$patient_obj.FullName", false] },
+                  then: "$patient_obj.FullName",
+                  else: {
+                    $concat: [
+                      { $ifNull: ["$patient_obj.FirstName", ""] },
+                      " ",
+                      { $ifNull: ["$patient_obj.LastName", ""] },
+                    ],
+                  },
+                },
+              },
+            },
+            patientPhone: { $first: { $ifNull: ["$patient_obj.Phone", ""] } },
+            lastMessageContent: { $first: { $ifNull: ["$content", ""] } },
+            lastMessageDirection: { $first: "$direction" },
+            lastMessageSentAt: { $first: "$sent_at" },
+          },
+        },
+        { $sort: { lastMessageSentAt: -1 } },
+        {
+          $facet: {
+            data: [{ $skip: skip }, { $limit: limitNum }],
+            count: [{ $count: "total" }],
+          },
+        },
+      ];
+
+      const [result] = await CommunicationHistory.aggregate(pipeline);
+
+      const conversations = result?.data ?? [];
+      const total = result?.count?.[0]?.total ?? 0;
+
+      // Enrich with upcoming follow-ups
+      const patientIds = conversations.map((c: any) => c._id);
+      const enrichedData = await this.enrichWithFollowUps(conversations, patientIds, userEmail);
+
+      return res.status(200).json({
+        status: true,
+        data: enrichedData,
+        pagination: {
+          page: pageNum,
+          limit: limitNum,
+          total,
+          totalPages: Math.ceil(total / limitNum),
+        },
+      });
+    } catch (error: any) {
+      appLogger.error("MessageConversationController/getConversations error:", error?.message);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };

+          followUpDate: { $gte: now },
+        },
+      },
+      { $sort: { followUpDate: 1 } },
+      {
+        $group: {
+          _id: "$patientId",
+          followUpId: { $first: "$_id" },
+          followUpDate: { $first: "$followUpDate" },
+          followUpStatus: { $first: "$status" },
+          followUpNotes: { $first: "$notes" },
+        },
+      },
+    ]);

+    const followUpMap = new Map(followUps.map((f: any) => [f._id, f]));

+      const fu = followUpMap.get(conv._id);

+        patientId: conv._id,
+        patientName: conv.patientName?.trim() || "",
+        patientPhone: conv.patientPhone,
