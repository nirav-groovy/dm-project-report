+export class DeviceRegistrationController {
+  private static instance: DeviceRegistrationController | null = null;
+
+  public static getInstance() {
+    DeviceRegistrationController.instance ??= new DeviceRegistrationController();
+    return DeviceRegistrationController.instance;
+  }

+  private readonly deviceRegistrationService: DeviceRegistrationService;
+
+  constructor() {
+    this.deviceRegistrationService = DeviceRegistrationService.getInstance();

+  /**
+   * Register or update a device (upsert). Called on user login.
+   * POST /device/register
+   * Body: { deviceToken, platform, deviceId }
+   */
+  upsertDevice = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { deviceToken, platform, deviceId } = req.body;
+      if (!deviceToken || !platform || !deviceId) {
+        return res.status(400).json({
+          status: false,
+          message: "deviceToken, platform, and deviceId are required",
+        });
+      }
+
+      if (!Object.values(DevicePlatform).includes(platform)) {
+        return res.status(400).json({
+          status: false,
+          message: `platform must be one of: ${Object.values(DevicePlatform).join(", ")}`,
+        });
+      }
+
+      const result = await this.deviceRegistrationService.upsertDevice({
+        userId,
+        deviceToken,
+        platform,
+        deviceId,
+      });
+
+      return res.status(200).json({ status: true, data: result });
+    } catch (error) {
+      appLogger.error("DeviceRegistrationController/upsertDevice error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+
+  /**
+   * Deactivate a specific device on logout.
+   * POST /device/deactivate
+   * Body: { deviceId }
+   */
+  deactivateDevice = async (req: UserRequest, res: Response) => {
+    try {
+      const userId = req.user?._id;
+      if (!userId) {
+        return res.status(401).json({ status: false, message: "User not authenticated" });
+      }
+
+      const { deviceId } = req.body;
+      if (!deviceId) {
+        return res.status(400).json({ status: false, message: "deviceId is required" });
+      }
+
+      const result = await this.deviceRegistrationService.deactivateDevice(userId, deviceId);
+      return res.status(200).json({ status: true, data: result });
+    } catch (error) {
+      appLogger.error("DeviceRegistrationController/deactivateDevice error:", error);
+      return res.status(500).json({ status: false, message: "Internal server error" });
+    }
+  };
+
+  /**
+   * Deactivate all devices for the current user.

+  deactivateAllDevices = async (req: UserRequest, res: Response) => {

+import { Schema, Document, Model, model } from "mongoose";
+import { DevicePlatform, DeviceRegistrationBody } from "../types/device_registration";

+/**
+ * Interface representing a device registration document.
+ */
+type IDeviceRegistration = DeviceRegistrationBody & Document;
+
+const DeviceRegistrationSchema: Schema<IDeviceRegistration> = new Schema<IDeviceRegistration>(
+  {
+    userId: {
+      required: true,
+      type: Number,
+      index: true,
+    },
+    deviceToken: {
+      required: true,
+      type: String,
+    },
+    platform: {
+      required: true,
+      type: String,
+      enum: Object.values(DevicePlatform),
+    },
+    deviceId: {
+      required: true,
+      type: String,
+    },
+    isActive: {
+      required: false,
+      type: Boolean,
+      default: true,
+    },
+  },
+  {
+    versionKey: false,
+    collection: "device_registrations",
+    timestamps: true,
+  }
+);
+
+// Compound index: one device per user (upsert key)
+DeviceRegistrationSchema.index({ userId: 1, deviceId: 1 }, { unique: true });
+
+export const DeviceRegistration: Model<IDeviceRegistration> = model<IDeviceRegistration>(
+  "DeviceRegistration",
+  DeviceRegistrationSchema
+);

+import { DeviceRegistration } from "../models";
+import { DevicePlatform, DeviceRegistrationResponse } from "../types/device_registration";
+import appLogger from "../utils/logger";
+
+/**
+ * Repository class for managing device registrations.
+ */
+export class DeviceRegistrationRepository {
+  private static instance: DeviceRegistrationRepository | null = null;
+
+  public static getInstance() {
+    if (!DeviceRegistrationRepository.instance) {
+      DeviceRegistrationRepository.instance = new DeviceRegistrationRepository();
+    }
+    return DeviceRegistrationRepository.instance;
+  }
+
+  /**
+   * Upsert a device registration. If the userId+deviceId combo exists, update the token;
+   * otherwise create a new record.
+   */
+  async upsert(data: {
+    userId: number;
+    deviceToken: string;
+    platform: DevicePlatform;
+    deviceId: string;
+  }): Promise<DeviceRegistrationResponse | null> {
+    try {
+      return await DeviceRegistration.findOneAndUpdate(
+        { userId: data.userId, deviceId: data.deviceId },
+        {
+          $set: {
+            deviceToken: data.deviceToken,
+            platform: data.platform,
+            isActive: true,
+          },
+        },
+        { upsert: true, new: true }
+      );
+    } catch (error: any) {
+      appLogger.error("DeviceRegistrationRepository/upsert error -->", error);
+      return null;
+    }
+  }
+
+  /**
+   * Deactivate a specific device for a user.
+   */
+  async deactivateDevice(userId: number, deviceId: string): Promise<DeviceRegistrationResponse | null> {
+    try {
+      return await DeviceRegistration.findOneAndUpdate(
+        { userId, deviceId },
+        { $set: { isActive: false } },
+        { new: true }
+      );
+    } catch (error: any) {
+      appLogger.error("DeviceRegistrationRepository/deactivateDevice error -->", error);
+      return null;
+    }
+  }
+
+  /**
+   * Deactivate all devices for a user (e.g., on full logout).
+   */
+  async deactivateAllByUserId(userId: number): Promise<number> {
+    try {
+      const result = await DeviceRegistration.updateMany(
+        { userId, isActive: true },
+        { $set: { isActive: false } }
+      );
+      return result.modifiedCount;
+    } catch (error: any) {
+      appLogger.error("DeviceRegistrationRepository/deactivateAllByUserId error -->", error);
+      return 0;
+    }
+  }
+
+  /**
+   * List all active devices for a user.
+   */
+  async findActiveByUserId(userId: number): Promise<Array<DeviceRegistrationResponse> | null> {
+    try {
+      return await DeviceRegistration.find({ userId, isActive: true }).sort({ updatedAt: -1 });
+    } catch (error: any) {
+      appLogger.error("DeviceRegistrationRepository/findActiveByUserId error -->", error);
+      return null;
+    }
+  }
+}
