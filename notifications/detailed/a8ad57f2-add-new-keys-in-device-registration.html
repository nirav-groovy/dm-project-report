+import { Request, Response, NextFunction } from "express";
+import { check, validationResult, ValidationChain } from "express-validator";
+import { DevicePlatform } from "../../types/device_registration";
+
+/**
+ * Represents an item containing validation result information.
+ * @interface ValidationResultItem
+ */
+interface ValidationResultItem {
+  msg: string;
+}
+
+/**
+ * Middleware function array to validate device registration data.
+ * @type {ValidationChain[]}
+ */
+export const deviceRegistrationValidation: ValidationChain[] = [
+  check("deviceToken").trim().notEmpty().withMessage("Device token is required"),
+  check("platform")
+    .trim()
+    .notEmpty()
+    .withMessage("Platform is required")
+    .isIn(Object.values(DevicePlatform))
+    .withMessage(`Platform must be one of: ${Object.values(DevicePlatform).join(", ")}`),
+  check("deviceId").trim().notEmpty().withMessage("Device ID is required"),
+  check("deviceName").trim().notEmpty().withMessage("Device name is required"),
+  check("appVersion").trim().notEmpty().withMessage("App version is required"),
+];
+
+/**
+ * Middleware function array to validate device deactivation data.
+ * @type {ValidationChain[]}
+ */
+export const deviceDeactivationValidation: ValidationChain[] = [
+  check("deviceId").trim().notEmpty().withMessage("Device ID is required"),
+];
+
+/**
+ * Middleware function for request validation.
+ * Validates the request using express-validator library and returns JSON response in case of validation errors.
+ * @param {Request} req - The Express request object.
+ * @param {Response} res - The Express response object.
+ * @param {NextFunction} next - The Express next function.
+ * @returns {Response|void} Returns a JSON response if validation fails, otherwise calls the next middleware.
+ */
+export const validation = (req: Request, res: Response, next: NextFunction): Response | void => {
+  let error: string[] = [];
+  const result: ValidationResultItem[] = validationResult(req).array();
+  if (!result.length) return next();
+
+  result.forEach((validationResultItem) => {
+    error.push(validationResultItem.msg);
+  });
+
+  return res.status(400).json({
+    status: false,
+    message: error.join(", "),
+    data: {},
+    code: 400,
+  });
+};
