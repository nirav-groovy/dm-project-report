+  private async initApp(): Promise<admin.app.App | null> {
+    try {
+      const env = process.env.NODE_ENV ?? "local";
+      const secret = await this.commonUtils.getSecret(`${env}-FIREBASE-SERVICE-ACCOUNT`);
+      if (!secret) {
+        appLogger.error("FcmUtils/initApp ~ Firebase service account secret is empty");
+        return null;
+      }
+
+      const serviceAccount = typeof secret === "string" ? JSON.parse(secret) : secret;
+      const existing = admin.apps.find((a) => a?.name === "comm-module-fcm");
+      if (existing) return existing;
+
+      return admin.initializeApp(
+        {
+          credential: admin.credential.cert(serviceAccount),
+        },
+        "comm-module-fcm"
+      );
+    } catch (error: any) {
+      appLogger.error("FcmUtils/initApp error:", error?.message ?? error);
+      return null;
+    }
+  }

+  /**
+   * Send a push notification to one or more device tokens.
+   * Returns per-call counts and the list of tokens FCM rejected as invalid
+   * so the caller can mark those device records as expired.
+   */
+  async sendToTokens(tokens: string[], payload: FcmPayload): Promise<FcmSendResult> {
+    const empty: FcmSendResult = { successCount: 0, failureCount: 0, invalidTokens: [] };
+    if (!tokens?.length) return empty;
+
+    const app = await this.getApp();
+    if (!app) return empty;
+
+    const message: admin.messaging.MulticastMessage = {
+      tokens,
+      notification: {
+        title: payload.title,
+        body: payload.body,
+      },
+      data: payload.data,
+      android: {
+        priority: "high",
+        notification: {
+          sound: "default",
+          channelId: "default",
+        },
+      },
+      apns: {
+        payload: {
+          aps: {
+            sound: "default",
+            badge: 1,
+            contentAvailable: true,
+          },
+        },
+      },
+    };
+
+    try {

+        failureCount: response.failureCount,

+    } catch (error: any) {
+      appLogger.error("FcmUtils/sendToTokens error:", error?.message ?? error);
+      return empty;
+    }
+  }
